42  public class FilterMappingsInstruction implements Instruction {
43      private static final Logger log = LoggerFactory.getLogger(FilterMappingsInstruction.class);
44      private static final String FILTER_CLASS = "org.springframework.web.filter.DelegatingFilterProxy";
45      private static final String FILTER_NAME = "springSecurityFilterChain";
46      private static final List<String> URL_PATTERNS = Collections.singletonList("/*");
47      private static final List<WebXmlService.Dispatcher> DISPATCHERS = Collections.singletonList(WebXmlService.Dispatcher.REQUEST);
48      private static final Module MODULE = Module.SITE_WEBAPP;
49      private static final Map<String, String> initParams = new HashMap<>();
50      public static final String HST_FILTER = "HstFilter";
51  
52  
53      @Inject
54      private WebXmlService webXmlService;
55  
56      @Inject
57      private ProjectService projectService;
58      @Override
59      public Status execute(final Map<String, Object> parameters) {
60          try {
61              // filter
62              webXmlService.addFilter(MODULE, FILTER_NAME, FILTER_CLASS, initParams);
63              webXmlService.insertFilterMapping(MODULE, FILTER_NAME, URL_PATTERNS, HST_FILTER);
64              webXmlService.addDispatchersToFilterMapping(MODULE, HST_FILTER, DISPATCHERS);
65              webXmlService.addDispatchersToFilterMapping(MODULE, FILTER_NAME, DISPATCHERS);
66              // context
67              final File webXml = projectService.getWebInfPathForModule(MODULE).resolve("web.xml").toFile();
68              final Document doc = new SAXReader().read(webXml);
69              final Node node = doc.selectSingleNode("/web-app");
70  
71              if (node instanceof Element) {
72                  final Element element = (Element) node;
73                  final Namespace namespace = element.getNamespace();
74                  final Element context = new BaseElement("context-param", namespace);
75                  final Element paramName = new BaseElement("param-name", namespace);
76                  paramName.setText("contextConfigLocation");
77                  context.add(paramName);
78                  final Element paramValue = new BaseElement("param-value", namespace);
79                  paramValue.setText("\n      /WEB-INF/applicationContext.xml\n" +
80                          "      /WEB-INF/applicationContext-security.xml");
81                  context.add(paramValue);
82  
83                  // listener
84                  final Element listener = new BaseElement("listener", namespace);
85                  final Element clazzElement = new BaseElement("listener-class", namespace);
86                  clazzElement.setText("org.springframework.web.context.ContextLoaderListener");
87                  listener.add(clazzElement);
88                  //
89                  element.elements().add(3, listener);
90                  element.elements().add(3, context);
91  
92  
93              }
94              Utils.prettyPrint(webXml, doc.asXML().getBytes());
95  
96  
97          } catch (Exception e) {
98              log.error("Error executing file mappings instruction:", e);
99          }
100 
101 
102         return Status.SUCCESS;
103     }
104 
105     @Override
106     public void populateChangeMessages(final BiConsumer<Type, String> changeMessageQueue) {
107         changeMessageQueue.accept(Type.EXECUTE, "Install Spring Security filter into Site web.xml.");
108     }
109 }
