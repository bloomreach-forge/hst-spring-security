16  package org.onehippo.forge.security.support.springsecurity.container;
17  
18  import java.security.Principal;
19  import java.util.HashSet;
20  import java.util.Set;
21  
22  import javax.jcr.Credentials;
23  import javax.jcr.SimpleCredentials;
24  import javax.security.auth.Subject;
25  import javax.servlet.http.HttpServletRequest;
26  import javax.servlet.http.HttpSession;
27  
28  import org.apache.commons.lang.StringUtils;
29  import org.hippoecm.hst.container.valves.AbstractOrderableValve;
30  import org.hippoecm.hst.core.container.ContainerConstants;
31  import org.hippoecm.hst.core.container.ContainerException;

37  import org.slf4j.LoggerFactory;

39  import org.springframework.security.core.GrantedAuthority;
40  import org.springframework.security.core.context.SecurityContext;

44  /**
45   * HST-2 request processing valve for integration with Spring Security Framework.
46   * <P>
47   * This is responsible for reading the <code>org.springframework.security.core.Authentication</code> instance if exists,
48   * and establishing <code>javax.security.auth.Subject</code> for the whole HST-2 request processing
49   * by converting <code>org.springframework.security.core.userdetails.UserDetails</code> into a set of
50   * <code>java.security.Principal</code>s (user principal and role principals from the collection of 
51   * <code>org.springframework.security.core.GrantedAuthority</code>).
52   * </P>
53   */
54  public class SpringSecurityValve extends AbstractOrderableValve {
55  
56      private static Logger log = LoggerFactory.getLogger(SpringSecurityValve.class);
57  
58      private static final char [] DUMMY_CHARS = { 'D', 'U', 'M', 'M', 'Y' };
59  
60      /**
61       * Flag whether or not to store JCR credentials as Subject's private credentials
62       * simply by converting username/password to <code>javax.jcr.SimpleCredentials</code> object.
63       */
64      private boolean storeSubjectRepositoryCredentials = true;
65  
66      /**
67       * Returns true if the option to store JCR credentials as Subject's private credentials is turned on.
68       * @return true if the option to store JCR credentials as Subject's private credentials is turned on
69       */
70      public boolean isStoreSubjectRepositoryCredentials() {
71          return storeSubjectRepositoryCredentials;
72      }
73  
74      /**
75       * Sets the flag whether or not to store JCR credentials as Subject's private credentials.
76       * @param storeSubjectRepositoryCredentials flag whether or not to store subject repository credentials
77       */
78      public void setStoreSubjectRepositoryCredentials(boolean storeSubjectRepositoryCredentials) {
79          this.storeSubjectRepositoryCredentials = storeSubjectRepositoryCredentials;
80      }
81  
82      @Override
83      public void invoke(ValveContext context) throws ContainerException {
84          HttpServletRequest request = context.getServletRequest();
85          Principal userPrincipal = request.getUserPrincipal();
86  
87          // If user has not been authenticated yet by any mechanism, then simply move to the next valve chain.
88          if (userPrincipal == null) {
89              if (log.isDebugEnabled()) {
90                  log.debug("No user principal found. Skipping SpringSecurityValve...");
91              }
92              context.invokeNext();
93              return;
94          }
95  
96          // Get the current subject from http session if exists.
97          HttpSession session = request.getSession(false);
98          Subject subject = (session != null ? (Subject) session.getAttribute(ContainerConstants.SUBJECT_ATTR_NAME)
99                  : null);
100 
101         // If a subject has been established already (normally by HST-2's SecurityValve), then simply move to the next valve chain.
102         if (subject != null) {
103             if (log.isDebugEnabled()) {
104                 log.debug("Already subject has been created somewhere before. Skipping SpringSecurityValve...");
105             }
106             context.invokeNext();
107             return;
108         }
109 
110         // Get Spring Security Context object from thread local.
111         SecurityContext securityContext = SecurityContextHolder.getContext();
112 
113         // If there's no Spring Security Context object, then just move to next valve chain.
114         if (securityContext == null) {
115             if (log.isDebugEnabled()) {
116                 log.debug("Spring Security hasn't established security context. Skipping SpringSecurityValve...");
117             }
118             context.invokeNext();
119             return;
120         }
121 
122         // Get the Authentication object from the Spring Security context object.
123         Authentication authentication = securityContext.getAuthentication();
124 
125         // If there's no Authentication object, it's really weird, so leave warning logs, and move to next valve chain.
126         if (authentication == null) {
127             if (log.isWarnEnabled()) {
128                 log.warn("Spring Security hasn't establish security context with authentication object. Skipping SpringSecurityValve...");
129             }
130             context.invokeNext();
131             return;
132         }
133 
134         // Get principal object from the Spring Security authentication object.
135         Object springSecurityPrincipal = authentication.getPrincipal();
136 
137         // We expect the principal is instance of UserDetails. Otherwise, let's skip it and leave warning logs.
138         if (!(springSecurityPrincipal instanceof UserDetails)) {
139             if (log.isWarnEnabled()) {
140                 log.warn("Spring Security hasn't establish security context with UserDetails object. We don't support non UserDetails authentication. Skipping SpringSecurityValve...");
141             }
142             context.invokeNext();
143             return;
144         }
145 
146         // Cast principal instance to UserDetails 
147         UserDetails userDetails = (UserDetails) springSecurityPrincipal;
148 
149         // Create HST-2 TransientUser principal from the user principal.
150         User user = new TransientUser(userPrincipal.getName());
151 
152         // Add both the existing user principal and new HST-2 user transient user principal
153         // just for the case when HST-2 can inspect the user principals for some reasons.
154         Set<Principal> principals = new HashSet<Principal>();
155         principals.add(userPrincipal);
156         principals.add(user);
157 
158         // Retrieve all the granted authorities from the UserDetail instance
159         // and convert it into HST-2 TransientRoles.
160         for (GrantedAuthority authority : userDetails.getAuthorities()) {
161             String authorityName = authority.getAuthority();
162             if (!StringUtils.isEmpty(authorityName)) {
163                 principals.add(new TransientRole(authorityName));
164             }
165         }
166 
167         Set<Object> pubCred = new HashSet<Object>();
168         Set<Object> privCred = new HashSet<Object>();
169 
170         // If the flag is turned on, then store JCR credentials as well
171         // just for the case the site is expected to use session stateful JCR sessions per authentication.
172         if (storeSubjectRepositoryCredentials) {
173             Credentials subjectRepoCreds = null;
174 
175             // Note: password should be null by default from some moment after Spring Security version upgraded a while ago.
176             //       if password is null, let's store a dummy password instead.
177 
178             if (userDetails.getPassword() != null) {
179                 subjectRepoCreds = new SimpleCredentials(userDetails.getUsername(), userDetails.getPassword()
180                         .toCharArray());
181             } else {
182                 subjectRepoCreds = new SimpleCredentials(userDetails.getUsername(), DUMMY_CHARS);
183             }
184 
185             privCred.add(subjectRepoCreds);
186         }
187 
188         subject = new Subject(true, principals, pubCred, privCred);
189 
190         // Save the created subject as http session attribute which can be read by HST-2 SecurityValve in the next valve chain.
191         request.getSession(true).setAttribute(ContainerConstants.SUBJECT_ATTR_NAME, subject);
192 
193         context.invokeNext();
194     }
